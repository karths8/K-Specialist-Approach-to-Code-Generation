SemDeDup: 

- Setup conda: https://docs.conda.io/projects/conda/en/latest/user-guide/install/linux.html
mkdir -p ~/miniconda3
wget https://repo.anaconda.com/miniconda/Miniconda3-latest-Linux-x86_64.sh -O ~/miniconda3/miniconda.sh
bash ~/miniconda3/miniconda.sh -b -u -p ~/miniconda3
rm -rf ~/miniconda3/miniconda.sh
~/miniconda3/bin/conda init bash
~/miniconda3/bin/conda init zsh

- Clone repo
git clone https://github.com/karths8/CS762_Project.git

- Setup conda env inside SemDeDup:
conda env create -n semdedup --file semdedup_conda_env.yml 
conda activate semdedup

- Clone codealpaca
git clone https://github.com/sahil280114/codealpaca.git

- Install dependencies
pip install transformers
pip install sentence-transformers
conda uninstall -c pytorch faiss-cpu
conda install -c pytorch faiss-gpu


- Run
python compute_pretrained_embeddings.py

python compute_centroids.py

python assign_and_sort_clusters.py
